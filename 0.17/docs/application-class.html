Application class

Для инициализации приложения необходимо:

  1. создать объект core.Application, передав в него объект xconfig (доступен в глобальном пространстве имен)
  2. задать в приложении доменную модель - это обычно объект, получаемый как импортируемый модуль "app/model-ext"
  3. задать в приложении реализацию DataFacade
  4. вызывать метод Application.run

Объект Application поддерживает кастомизацию - можно переопределить методы (перед вызовом run):

  • preinitialize
  • initialize
  • postinitialize

Методы будут вызваны в рамках инициализации объекта Application, запущенной методом run:

  • вызов onAppStateChanged("loading")
  • инициализация арий (поиск элементов с css-классов x-area)
  • установка экземпляра DataFacade, если он не был задан явно: при подключении модуля "lib/interop/DataFacade" и "lib/interop/BackendInterop" и отсутствии экземпляра в поле Application.dataFacade, будет автоматически создан экземпляр DataFacade (реализация "lib/interop/DataFacade" поддерживает только online работу. Для поддержки работы в offline необходимо использовать реализацию "lib/interop/DataFacadeSmart", подробней см. в статье про offline)
  • вызов preinitialize
  • инициализация App-модулей, подключенных как зависимости RequireJS-модуля в main.js
  • вызов initialize
  • вызов onAppStateChanged("initialized")
  • если initialize вернул Deferred, то дальнейшее выполнение откладывается до момента его срабатывания (resolve)
  • запуск AppStateManager (вызов его start)
  • вызов postinitialize - базовая реализация публикует событие "app.start"
  • вызов onAppStateChanged("started")
  • в случае ошибки вызовается onAppStateChanged("failed")

Метод onAppStateChanged по умолчанию для каждого состояния приложения вызывается соответствующий метод:

  • "loading" - onLoading: function () {},
  • "initialized" - onInitialized: function () {},
  • "started" - onStarted: function () {},
  • "failed" - onFailed: function (error) {}

Прикладное приложение может как реализовать методы onXXX, так и заменить onAppStateChanged. Основное предназначение onAppStateChanged - отображение процесса загрузки приложения. См. пример в App starting UX.

Примеры реализация app.js

типовая реализация app/app.js
define([
    "jquery",
    "core",
    "app/model-ext",
    "lib/interop/DataFacadeSmart",
    "lib/interop/BackendInterop",
    "app/CacheManager",
    "lib/data/DataStore",
    "lib/ui/menu/NavMenu",
    "lib/ui/menu/LanguageMenu",
    "lib/ui/menu/SystemMenu",
    "lib/ui/OnlineBeacon",
    "i18n!app/nls/resources",
    "app/iconProvider",
    "app/default-menus",
    "app/default-parts",
    "app/metadata",
    "app/metadata-survey",
    "app/module-main",
    "app/module-about",
    "app/module-admin",
    "app/module-breadcrumbs"
],
function ($, core, model, DataFacade, BackendInterop, CacheManager, DataStore, NavMenu, LanguageMenu, SystemMenu, OnlineBeacon, resources) {
    "use strict";
    var app = new core.Application(xconfig),
        store = new DataStore(xconfig.appName, 3, model.meta);
    app.model = model;
    app.dataFacade = new DataFacade(new BackendInterop(xconfig), app.eventPublisher, store, {
        cacheManager: new CacheManager()
    });

    app.preinitialize = function () {
        var that = this;
        that.navMenu = new NavMenu(that.areaManager);
        that.sysMenu = new SystemMenu(that);
    };

    app.initialize = function () {
        var that = this;                       

        // установка заголовков Арий (можно определить в html)
        that.areaManager.getArea(null).title = resources["area.main"];

        that.navMenu.render($("#areasMenu"));

        // системное меню (в частности Черновики)
        that.sysMenu.render($("#sysMenu"));

        // authentication menu (login/logout)
        // NOTE: AuthMenu can't be created inside preinitialize as it uses getCurrentUser
        that.authMenu = new core.ui.AuthMenu(that.dataFacade, that.eventPublisher);
        that.authMenu.render($("#account-menu"));

        // меню для переключения текузего языка
        that.langMenu = new LanguageMenu(that.config);
        that.langMenu.render($("#lang-menu"));

        // отображение состония offline-приложения
        that.onlineBeacon = new OnlineBeacon(that);
        that.onlineBeacon.render($("#online-beacon"));
    };

    return app;
});
