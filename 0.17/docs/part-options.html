Part options

Зачастую при создании парта требуется указывать дополнительные параметры или опции. Как правило для этих целей в конструктор парта добавляют параметр options. Это plain javascript object, содержащий необходимые значения опций. Обычно внутри конструктора этот параметр запоминается в поле this.options для дальнейшего использования. В примитивном случае все просто:

var MyPart = core.lang.Class(Part, {
    constructor: function (options) {
        var that = this;
        Part.call(that);
        that.options = options;
        // other initialization
    }
});

Но есть несколько проблем:

  • параметр options может быть не задан
  • не все свойства в options могут заданы, в этом случае нужно использовать какие-то значения по умолчанию
  • если парт участвует в иерархии наследования, то все еще сложнее:
    • в базовый конструктор нужно передавать корректные значения опций
    • нужно учитывать значения опций по умолчанию для всех классов в иерархии

Чтобы решить все эти проблемы в базовом классе Part введен метод mixOptions. Реализация сводится к следующиему:

mixOptions: function (options, defaultOptions) {
    // some argument validation
    return lang.deepAppend(this.options 
| options  |
|- | {}, defaultOptions);
}

Как видно метод весьма примитивен. Но его при его использовании нужно соблюдать некоторые правила:

  • метод должен вызываться до вызова родительского конструктора
  • в прототипе класса не должно быть поля options
  • после вызова mixOptions в конструкторе использовать поле this.options, а не значение параметра options
  • если родительский парт использует механизм mixOptions, то все дочерние парты также должны его использовать
  • опции по умолчанию рекомендуется (хотя и не обязательно) объявлять в поле defaultOptions прототипа класса. Если вы следуете этому правилу, то при вызове mixOptions обращайтесь к этому полю через prototype, не используя this. Иначе возможно некорректное поведение при наследовании.

Bсе парты в составе XFW3 WebClient используют механизм mixOptions, начиная с версии 0.9. Если вы создаете наследника от стандартного парта, то вы также должны использовать mixOptions.

Пример правильного испольвания mixOptions:

var ParentPart = core.lang.Class(Part, {
    defaultOptions: {
        option1: "parent1",
        option2: "parent2"
    },
    constructor: function (options) {
        var that = this;
        that.options = that.mixOptions(options, ParentPart.prototype.defaultOptions);
        Part.call(that);
        // other initialization, e.g.
        if (that.options.option1 === "some value1") { /* do something */ }
        if (that.options.option2 === "some value2") { /* do something */ }
    }
});
var ChildPart = core.lang.Class(ParentPart, {
    defaultOptions: {
        option2: "child2", // override parent default value
        option3: "child3"  // specific child option
    },
    constructor: function (options) {
        var that = this;
        that.options = that.mixOptions(options, ChildPart.prototype.defaultOptions);
        ParentPart.call(that, options);
        // other initialization, e.g.
        if (that.options.option3 === "some value3") { /* do something */ }
    }
});

Определение defaultOptions в прототипе удобно и позволяет легко изменять опции партов по умолчанию, например:

ParentPart.prototype.defaultOptions.option1 = "custom value";

Пример неправильного испольвания mixOptions:

var BasePart = core.lang.Class(Part, {
    defaultOptions: {
        option1: "parent"
    },
    constructor: function (options) {
        var that = this;
        that.options = that.mixOptions(options, ParentPart.prototype.defaultOptions);
        // Здесь options.option1 будет === "parent", а не "child", 
        Part.call(that);
    }
});
var CustomPart = core.lang.Class(BasePart, {
    defaultOptions: {
        option2: "child"
    },
    constructor: function (options) {
        var that = this;
        that.options = that.mixOptions(options, CustomPart.prototype.defaultOptions);
        BasePart.call(that, {
            options1: "child"
        });
    }
});

В констуркторе BasePart после вызова minOptions значение в this.options.option1 будет равно "parent", а не "child", как можно было бы ожидать. Все из-за того, что mixOptions мерджил текущий this.options и BasePart.prototype.defaultOptions, а параметр options не использовался. Т.е. если производный парт присваивает поле options, то базовых классах будет использоваться именно оно, а не параметр конструктора (см. реализацию метода mixOptions). Чтобы не запутаться параметр базового констурктора можно не задавать:

var CustomPart = core.lang.Class(BasePart, {
    defaultOptions: {
        option2: "child"
    },
    constructor: function (options) {
        var that = this;
        that.options = that.mixOptions(options, CustomPart.prototype.defaultOptions);
        that.options.options1 = "child";
        BasePart.call(that);
    }
});
