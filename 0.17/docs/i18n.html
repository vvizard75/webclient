i18n

Client

Локализация пользовательского интерфейса основана на плагине i18n для RequireJS. Все строки должны браться из файла ресурсов, который импортируется с помощью этого плагина. Пример:

define([
    "core",
    "i18n!app/nls/resources",
], function (core, resources) {
    "use strict";

    core.createAreaModule(function (app, area) {
        var menuRegion = area.regionManager.getRegion("mainmenu");

        area.addState({name: "surveys", title: resources["surveys"] }, {"main": "surveys"});
    });
});

Файл "app/nls/resources":

define({
    "root": {
        "surveys": "Surveys"
    },
    "ru": true
});

Рядом с ним в поддиректории должен быть файл для языка "ru" - "app/nls/ru/resources":

define({
    "surveys": "Опросы"
});

Подробней о i18n.js см. в документации.

Текущий язык плагин j18n.js определяет на основании свойства locale конфигурации RequireJS. Помимо этого текущий язык приложения хранится в свойстве core.platform.language(). При изменении значения свойства, новое значение устанавливается в куку "X-Lang" и страница перегружается. При обновлении страницы на сервере значение из куки будет установлено в свойство locale конфигурации RequireJS. Таким образом можно изменять язык приложения.

Для переключения языка приложения предназначен компонент LanguageMenu.

var app = new core.Application(xconfig);
    app.initialize = function () {
        that.langMenu = new LanguageMenu(that.config);
        that.langMenu.render($("#lang-menu"));
    };
    app.run();

Компонент LanguageMenu создает меню из доступных языков на основе свойства supportedLanguages конфигуарции (main.config.json):

"supportedLanguages": {
        "ru": {"title": "Русский"},
        "en": {"title": "English"}
    }

Выбранный язык устанавливается в core.platform.language().

Для многоязычных приложений все строки должны браться из ресурсов. Существует множество мест, откуда строки попадают в пользовательский интерфейс:

  • серверная разметка (_Layout.cshtml). Отсюда строки надо просто удалить и устанавливать значения на клиенте. Типой случай - это наименования (title) арий. Их надо устанавливать в методе Application.initialize:
that.areaManager.getArea(null).title = resources["area.main"];
            that.areaManager.getArea("admin").title = resources["area.admin"];
  • шаблоны Handlebars. Для вставки локализованных строк в шаблоны следует использовать хелпер res:
<label>{{res "yes"}}:</label>

где "yes" имя ресурса. Реализация хелпера тривиальна:

Handlebars.registerHelper("res", function (name) {
        return new Handlebars.SafeString(resources[name]);
    });

где resources - модуль, импортированный через плагин i18n.

  • метаданные. При генерации json-метаданных доменной модели из DslDesigner'a для всех типов и свойств задано описание (атрибут descr) так, как он был задан в дизайнере. Для локализации описаний свойств и типов необходимо:
    • в model-ext перед вызовом domain.buildModel вызвать domain.localize(metadata, resources), где metadata - это модуль "model.js", а resources - модуль, загружаемый плагином i18n с ресурсами
    • в модуле ресурсов добавить ресурс для каждого типа и свойства по следующим правилам:
      • префикс "model."
      • "model.ТипХХХ" - описание (descr) типа "ТипХХХ"
      • "model.ТипХХХ.свойствоYYY" - описание (descr) свойства "свойствоYYY" типа "ТипХХХ"
    • при этом, дублировать описания из модели для такого языка не обязательно, т.к. метод domain.localize меняет описание элемента модели только в случае, если ресурс существует
  • серверные сообщения. Локализация серверных сообщений должна происходит автоматически на основании культуры текущего потока (при условии конечно, что все сообщения берутся из ресурсов и существуют ресурсы для требуемой культуры)

Особенности в Offline-режиме

При изменении языка (установке куки) и перезагрузке страницы, если приложения загружено из AppCache (см. Offline), то реального обращения на сервер не будет - страница будет загружена из кеша. И как следствие вся конфигурация останется прежней. Поэтому для оффлайн приложения (использующий HTML5 AppCache) в main.js необходимо устанавливать locale для RequireJS из куки явно:

var xconfig = xconfig || { root: '', require: {} };
function parseCookies() {
    var cookieObject = {},
        cookiesArray = document.cookie ? document.cookie.split('; ') : [],
        i, group, idx, key;

    for (i = 0; i < cookiesArray.length, group = cookiesArray[i], idx = group.indexOf('='); i=i+1) {
        key = decodeURIComponent(group.substr(0, idx));
        cookieObject[key] || (cookieObject[key] = decodeURIComponent(group.substr(idx + 1)));
    }
    return cookieObject;
}
var lang = parseCookies()["X-Lang"];
if (lang) {
    xconfig.require.locale = lang;
}
require.config(xconfig.require);

Server

На сервер для Mvc-контроллеров и Api-контроллеров добавляются фильтры, которые определяют текущий язык пользователя. Полученный язык они устанавливают в контекст вызова (в словари HttpContextBase.Items для MVC и HttpRequestMessage.Properties для WebApi добавляет элемент с имененем XCurrentUserLanguageHelper.HeaderName) и в качестве текущей культуры потока (Thread.CurrentThread.CurrentCulture). Определение текущего языка происходит по следующему алгоритму:

  • имем куку "X-Lang", если есть, то используем ее значение
  • если нет, то берем XConfig.DefaultLanguage, если он есть
  • если DefaultLanguage не задан, то берем хидер Accept-Language http-запроса (если устанавливает браузер)
